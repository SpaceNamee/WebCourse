		<script>
			function FinalSalary(salary, insure) { 
				return salary - (salary * (insure/100));
			}

			function SalaryForYear(numberOfMonth, k) { 
				return k * numberOfMonth;
			}

			let result = SalaryForYear(4, FinalSalary(20000, 10));
